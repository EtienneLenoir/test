Public Function Access_Table_To_CSV( _
        ByVal accdbPath As String, _
        ByVal tableName As String, _
        ByVal csvPath As String) As Boolean

    On Error GoTo FAIL

    Dim acc   As Object          ' Access.Application
    Dim fso   As Object
    Dim folder As String

    ' 1) S'assurer que le dossier existe
    Set fso = CreateObject("Scripting.FileSystemObject")
    folder = fso.GetParentFolderName(csvPath)
    If Len(folder) > 0 Then
        If Not fso.FolderExists(folder) Then
            fso.CreateFolder folder
        End If
    End If

    ' 2) Effacer un éventuel ancien CSV
    On Error Resume Next
        Kill csvPath
    On Error GoTo FAIL

    ' 3) Lancer Access en late binding
    Set acc = CreateObject("Access.Application")
    acc.Visible = False
    acc.OpenCurrentDatabase accdbPath

    ' TransferText:
    '   0  = acExportDelim
    '   vbNullString = aucune spécification d'export personnalisée
    '   tableName    = nom de la table Access
    '   csvPath      = chemin du fichier de sortie
    '   True         = inclure les noms de colonnes
    acc.DoCmd.TransferText 0, vbNullString, tableName, csvPath, True

    acc.CloseCurrentDatabase
    acc.Quit
    Set acc = Nothing

    Access_Table_To_CSV = (Dir$(csvPath, vbNormal) <> vbNullString)
    Exit Function

FAIL:
    On Error Resume Next
    If Not acc Is Nothing Then
        If acc.CurrentDb Is Nothing Then
        Else
            acc.CloseCurrentDatabase
        End If
        acc.Quit
    End If
    Set acc = Nothing

    Access_Table_To_CSV = False
End Function


' Dans cDuck
Public Sub ImportAccessTableFast( _
        ByVal accdbPath As String, _
        ByVal accessTable As String, _
        ByVal duckTableFullName As String, _
        Optional ByVal replace As Boolean = True, _
        Optional ByVal deleteTemp As Boolean = True)

    Dim tmpCsv As String
    Dim ok     As Boolean

    ' 1) Fichier CSV temporaire
    tmpCsv = ThisWorkbook.Path & "\acc_tmp_" & accessTable & ".csv"
    ' si tu préfères %TEMP% :
    ' tmpCsv = Environ$("TEMP") & "\acc_tmp_" & accessTable & ".csv"

    ' 2) Export Access -> CSV
    ok = Me.Access_Table_To_CSV(accdbPath, accessTable, tmpCsv)
    If Not ok Then
        Err.Raise 5, , "Access_Table_To_CSV KO pour " & accessTable & _
                       " : " & Native_LastErrorText()
    End If

    ' 3) Import CSV -> DuckDB (via ta DLL, super rapide)
    If replace Then
        ' DROP + recrée la table (auto-typage DuckDB)
        Me.ImportCsvReplace tmpCsv, duckTableFullName
    Else
        ' CREATE IF NOT EXISTS + DELETE + INSERT (selon ton ImportCsvFast)
        Me.ImportCsvFast tmpCsv, duckTableFullName, True
    End If

    ' 4) Nettoyage fichier temporaire
    If deleteTemp Then
        On Error Resume Next
        Kill tmpCsv
        On Error GoTo 0
    End If
End Sub



Sub Demo_Import_Access_Clients()
    Dim duck As cDuck
    
    Set duck = CurrentDuckDb          ' ton singleton
    duck.OpenDuckDb ThisWorkbook.Path & "\data.duckdb"

    ' Access → CSV → DuckDB.main.clients (en remplaçant la table)
    duck.ImportAccessTableFast _
            accdbPath:="C:\Data\monfichier.accdb", _
            accessTable:="Clients", _
            duckTableFullName:="main.clients", _
            replace:=True

    duck.CloseDuckDb
End Sub
